<i> Fazer um programa para ler os dados de um funcionário (nome, salário bruto e imposto). Em seguida, mostrar os dados do funcionário (nome e salário líquido.
Em seguida, aumentar o salário do funcionário com base em uma porcentagem dada (somente o salário bruto pode ser afetado pela porcentagem inserida)
e mostrar novamente os dados do funcionário:

Classe: Funcionario
- Name: String
- SalarioBruto: double
- Imposto: double

---
- SalarioLiquido: double
- AumentarSalario(porcentagem: double) : void </i>

```
package entities;

public class Funcionario {
	
	public String nome;
	public double salarioBruto;
	public double imposto;
	public double porcentagem;
	
	
	public double calculoPorcentagem(double porcentagem, double salarioBruto) {
		return ( salarioBruto * porcentagem ) / 100 ;
	}
	
	public double getSalarioLiquido() {
		return salarioBruto - imposto;
	}
	
	public void aumentarSalario(double porcentagem) {
		salarioBruto += calculoPorcentagem(porcentagem, salarioBruto);
	}
	
	public String toString() {
		return "Funcionário: " 
				+ nome
				+ ", Salário bruto: "
				+ " R$"
				+ String.format("%.2f", salarioBruto)
				+ ", imposto: "
				+ String.format("%.2f", imposto)
				+ ", Salário líquido: "
				+ String.format("%.2f", getSalarioLiquido());
	}
	
}
```

```
package application;

import java.util.Scanner;

import entities.Funcionario;

public class Program {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		Funcionario funcionario = new Funcionario();
		
		System.out.println("Entre com o nome do funcionário: ");
		funcionario.nome = sc.nextLine();
		
		System.out.println("Entre com o salário bruto: ");
		funcionario.salarioBruto = sc.nextDouble();
		
		System.out.println("Entre com o imposto: ");
		funcionario.imposto = sc.nextDouble();
		
		System.out.println("Dados do funcionário: " + funcionario);
		
		
		System.out.println("Informe a porcentagem de aumento salarial: ");
		double perc = sc.nextDouble();
		funcionario.aumentarSalario(perc);
		
		System.out.println("Dados atualizados do funcionário: " + funcionario);
		
		
		
		sc.close();
	}
}
```
