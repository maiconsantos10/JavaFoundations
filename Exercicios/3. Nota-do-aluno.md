<i>Fazer um programa para ler o nome de um aluno e as três médias que ele obteve nos três trimestres do ano (primeiro trimestre vale 30 e o
segundo e terceiro valem 35 cada). Ao final, mostrar qual a nota final do aluno no ano. Dizer também se o aluno está aprovado (PASS) ou não (FAILED)
e, em caso negativo, quantos pontos faltam para o aluno obter o mínimo para ser aprovado (que é 60% da nota). 

Classe: Student
- Name: String
- mediaTrimestre1: double
- mediaTrimestre2: double
- mediaTrimestre3: double </i>

```
package entities;

public class Aluno {
	
	public String nome;
	
	public double mediaTrimestre1;
	public double mediaTrimestre2;
	public double mediaTrimestre3;
	
	public String toString() {
		return nome
				+ ", Primeira média: "
				+ mediaTrimestre1 
				+ ", segunda média: "
				+ mediaTrimestre2
				+ ", terceira média: "
				+ mediaTrimestre3 + "\n"
				+ "A média final do aluno é: "
				+ calcularMediaFinal();
	}
	
	public double calcularMediaFinal() {
		return mediaTrimestre1 + mediaTrimestre2 + mediaTrimestre3;		
	}
	
	public double pontosRestantes() {
		return 60 - calcularMediaFinal();
	}
	
	public void aprovado() {
		if ((calcularMediaFinal()) < 60) {
			System.out.println("STATUS FINAL - Este aluno está reprovado!");
			System.out.println();
			System.out.println("Pontos que faltaram para o aluno passar: " + pontosRestantes());
		} else {
			System.out.println("STATUS FINAL - PARABÉNS, VOCÊ FOI APROVADO!!!");
		}
	}
}
```

```
package application;

import java.util.Scanner;

import entities.Aluno;

public class Program {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		Aluno aluno = new Aluno();
		
		System.out.println("Insira o nome do aluno: ");
		aluno.nome = sc.nextLine();
		
		System.out.println("Insira a primeira média: ");
		aluno.mediaTrimestre1 = sc.nextDouble();
		
		System.out.println("Insira a segunda média: ");
		aluno.mediaTrimestre2 = sc.nextDouble();
		
		System.out.println("Insira a terceira média: ");
		aluno.mediaTrimestre3 = sc.nextDouble();
		
		System.out.println("Dados finais do aluno: " + aluno);
		
		System.out.println();
		aluno.aprovado();
		
		
		
		sc.close();
	}
}
```
